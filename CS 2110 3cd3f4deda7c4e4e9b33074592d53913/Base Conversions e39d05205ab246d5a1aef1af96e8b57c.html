// Does not include negative numbers
public static int binaryStringToInt(String binary) {
    int output = 0;
    for (int i = 0; i < binary.length(); i++) {
        if (binary.charAt(i) == '1') {
            output += 1 << binary.length() - i - 1;
        }
    }
    return output;
}

public static String binaryStringToOctalString(String binary) {
    int decimal = 0;
    for (int i = 0; i < binary.length(); i++) {
        if (binary.charAt(i) == '1') {
            decimal += 1 << binary.length() - i - 1;
        }
    }

    String output = "";
    int counter = 21;
    for (int i = 7; i >= 0; --i) {
        output += (decimal >> counter) & 0b111;
        counter -= 3;
    }
    return output;

public static String intToHexString(int hex) {
    String output = "";
    int counter = 16;
    while (counter <= hex) {
        counter = counter << 4;
    }
    counter = counter >> 4;
    int counter2 = 0;
    while (counter != 0) {
        if (hex < counter) {
            if (counter2 > 9) {
                output += (char)(counter2 + 65 - 10);
            } else {
                output += counter2;
            }
            counter = counter >> 4;
            counter2 = 0;
        }
        hex -= counter;
        counter2++;
    }
    return output;
}
